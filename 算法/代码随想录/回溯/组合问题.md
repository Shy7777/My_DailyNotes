class Solution:
	def findSubsequences(self, nums: List[int]) -> List[List[int]]:
		result=set()
		self.backTrace(nums,0,[],result)
		return [list(seq) for seq in result]
	def backTrace(self,nums,startIndex,path,result):
		len_num=len(path)
		if len_num>1 and path[len_num-1]<path[len_num-2]:
			return
		if len_num>1:
			result.add(tuple(path))
		for i in range(startIndex,len(nums)):
			path.append(nums[i])
			self.backTrace(nums,i+1,path,result)
			path.pop()
