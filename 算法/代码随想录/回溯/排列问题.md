46.全排列
不要startIndex了，反而使用used，如何正确使用used，见下述代码：
```python
class Solution:
	def permute(self, nums: List[int]) -> List[List[int]]:
		result=[]
		self.backTrace(nums,[],[False]*len(nums),result)
		return result
	def backTrace(self,nums,path,used,result):
		if len(path)==len(nums):
			result.append(path[:])
			return
		for i in range(len(nums)):
			if used[i]:
				continue
			used[i]=True
			path.append(nums[i])
			self.backTrace(nums,path,used,result)
			path.pop()
			used[i]=False
```
