    def removeElements(self, head, val):
        ## 创建虚拟头结点 这个头结点保存的是他与他之后的所有下一个结点的多维数组
        dummy_head = ListNode (next = head)
        current = dummy_head
        ## 循环不变量，当没有下一个结点即循环停止
        while (current.next):
            if (current.next.val==val):
                current.next = current.next.next
            else :
                current=current.next
        ## 返回所有结点
        return dummy_head.next
